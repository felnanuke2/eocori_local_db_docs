factory Field.fromMap(Map<String, dynamic> map) {
  return Field(
    idAutoIncrement: map['idAutoIncrement'],
    id: map['id'] ?? '',
    creation_time: DateTime.parse(map['creation_time']),
    update_time: map['update_time'] != null
        ? DateTime.parse(map['update_time'])
        : null,
    name: map['name'],
    description: map['description'],
    created_by: map['created_by'],
    updated_by: map['updated_by'],
    status: map['status']?.toInt() ?? 0,
    data_type: map['data_type']?.toInt() ?? 0,
    optionsIds: List.from(map['options'])
        .map((e) => FieldOption.fromMap(e).id)
        .toList(),
  );
}
