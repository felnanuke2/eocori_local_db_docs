FutureOr<Map<String, String>> authHeaders
override

Generate base auth headers for you requests

if isAuth is false the token of authHeaders will be empty or throws error

Implementation

@override
FutureOr<Map<String, String>> get authHeaders async =>
    _isar.writeTxn((isar) async {
      final token = await isar.accessTokens.where().findFirst();

      return {
        HttpHeaders.authorizationHeader:
            '${token?.tokeType ?? ''} ${token?.token ?? ''}'
      };
    });
