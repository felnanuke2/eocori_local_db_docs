factory FormEntry.fromMap(Map<String, dynamic> map) {
  final form = FormEntry(
      idAutoIncrement: map['idAutoIncrement']?.toInt(),
      id: map['id'] ?? '',
      creation_time: DateTime.parse(map['creation_time']),
      update_time: map['update_time'] != null
          ? DateTime.parse(map['update_time'])
          : null,
      name: map['name'],
      description: map['description'],
      created_by: map['created_by'],
      updated_by: map['updated_by'],
      status: map['status']?.toInt() ?? 0,
      statement: map['statement'],
      field_id: map['field_id'] ?? '',
      attachment_mode: map['attachment_mode'] ?? 0,
      linksIds: List.from(map['links'] ?? [])
          .map((e) => Link.fromMap(e).id)
          .toList());

  return form;
}
